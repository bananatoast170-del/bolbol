        <p class="text-slate-500 mb-8">כל לחיצה נחשבת!</p>

        <!-- תצוגת המונה -->
        <div class="mb-8">
            <span id="counter" class="text-7xl font-black text-blue-600 transition-all duration-300 inline-block">0</span>
            <div class="text-xs font-semibold text-slate-400 uppercase tracking-widest mt-2">לחיצות עד כה</div>
        </div>

        <!-- הכפתור הראשי -->
        <button id="clickBtn" class="click-animation glow-button w-full bg-blue-600 hover:bg-blue-700 text-white font-bold py-6 rounded-2xl text-2xl shadow-lg focus:outline-none focus:ring-4 focus:ring-blue-300">

            </p>
        </div>
    </div>

    <script>
        // אתחול המשתנים
        let count = 0;
        const counterElement = document.getElementById('counter');
        const button = document.getElementById('clickBtn');

        // פונקציה לעדכון המונה
        function updateCounter() {
            count++;
            counterElement.innerText = count;
            
            // הוספת אנימציה קטנה למספר בכל לחיצה
            counterElement.classList.add('scale-110');
            setTimeout(() => {
                counterElement.classList.remove('scale-110');
            }, 100);
        }

        // האזנה ללחיצות (עובד גם בעכבר וגם במגע בטלפון)
        button.addEventListener('click', updateCounter);

        // בונוס: שמירה מקומית (LocalStorage) כדי שהמספר לא יתאפס ברענון הדף אצל המשתמש
        // הערה: כדי שכל העולם יראה את אותו מספר בדיוק, צריך בסיס נתונים (Backend).
        window.onload = () => {
            const savedCount = localStorage.getItem('myClickCount');
            if (savedCount) {
                count = parseInt(savedCount);
                counterElement.innerText = count;
            }
        };

        button.addEventListener('click', () => {
            localStorage.setItem('myClickCount', count);
        });
    </script>
